// Teraform - Ec2 with Key Pair (c6i.8xlarge)

   provider "aws"  {
      region = "us-east-1"
   }

   resource "aws_key_pair" "PathnexKey" {
     key_name   = "PathnexKey"
     public_key = "ssh-rsa AAAA....."
   }


   resource "aws_instance" "PathnexEC2"
      ami            = "ami-0abcd1234abcd1234"
      instance_type  = "c6i.8xlarge" 
      key_name       = aws_key_pair.OathnexKey.key_name

//Parallel Builds
//Jenklins Pipeine - parallel Build and Test

pipeline {
    agent any
    stages {
        stage('Parallel Tasks') {
           parallel {
               stage('Build') {
                   steps {
                       sh 'mvn clean compile' 
                   }
                }
                stage('Test') {
                   steps {
                       sh 'mvn test'
                   }
                 }
               }
             }
           }
         }
