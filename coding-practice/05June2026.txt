#!/bin/bash

//Day 03 - User , 5Gs, Env Vars

//Ansible - create User 


---
- name: Create Pathnex user
  hosts: all
  become: yes

  tasks:
    
     - name: Create user Pathnex-admin
       user:
         name: pathnex-admin
         shell: /bin/bash


// Teraaform - Security Group (r6i.$xlarge Ec2)

provider "aws" {
   region = "us-east-1"

}


resource "aws_security_group" "Pathnex5G"
   name        = "PathnexSG"
   descripto   = "Pathnex SG for SSH"


   ingress {
     from_port    = 22
     to_port      = 22
     protocol     = "top"
     cidr_blocks  = {"0.0.0.0/0"}
    }
  }


  resource "aws_instance" "PathnexEC2" {
    ami                    = "ami-0abcd1234abcd1234"
    instance_type          = "r6i-4xlarge"
    vpc_security_group_ids = [aws_security_group.PathnexSG.id]
     

    tags = {
      Name = "Pathnex-EC2"
     }
   }





//Kubernetes - Ass Environment Variables 

apiVersion: v1
kind: pod
metadata:
   name: pathnex-env-pod

spec:
  container:
     - name: app
       image: nginx
       env:
         - name: APP_ENV
           value: "production"



// Shell Script -Check Files Exist

#!/bin/bash
FILE="/tmp/pathnex.txt"


if [ -f "$FILE" ]; then
  echo "File exists"
else
   echo "File does not exists"
fi




//Jenkins Pipeline - NodeJS Build

pipeline {
    agent any
    tools 'NodeJS-16'
    }
    stage('Checkout') {
       steps {
           git url: 'https://github.com/arjundalal1/sample-node-app.git
        }
    }
    stage('Install Dependencies') {
       steps {
           sh 'npm install'
       }
    }
    stage('Run Tests') {
        steps {
            sh 'npm test'
         }
    }
    stage('Build') {
         steps {
             sh 'npm run build'

         }
     }
  }
}
